Яке значення матимуть наступні вирази?

99  = 0110 0011
78  = 0100 1110
-13 = 1111 0011
177 = 1011 0001
7   = 0000 0111
255 = 1111 1111
240 = 1111 0000
42  = 0010 1010
-2  = 1111 1110

a) 99 and 78	// 0100 0010 = 66
b) 99 or 78		// 0110 1111 = 111
c) 99 xor 78	// 0010 1101 = 45
d) not 0		// 1111 1111 = -1
e) not -13		// 0000 1100 = 12
f) not 177		// 0100 1110 = 78
g) 1 shl 7		// 1000 0000 = 128
h) 7 shl 2		// 0001 1100 = 28
i) 255 shr 4	// 0000 1111 = 15
j) 240 shr 3	// 0001 1110 = 30 
k) 42 shr 3		// 0000 0101 = 5
l) -2 shl 2		// 1111 1000 = -8
