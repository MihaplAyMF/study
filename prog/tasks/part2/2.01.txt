Для кожного знаступних виразів визначте його значення та тип
a) 2+3				// = 5;		integer
b) 2+3*10 			// = 32;	integer
c) 5-(6-7)			// = 6;		integer
d) 18/6				// = 3.0;	real
e) 15/6				// = 2.5;	real
f) 18 div 6			// = 3;		integer
g) 18 mod 6			// = 0;		integer
h) 17 div 3			// = 5;		integer
i) 17 mod 3			// = 2;		integer
j) 4 mod 7			// = 4;		integer
k) 2/1 + 3/1		// = 5.0;	real
l) 3 < 4			// = true;	boolean
m) 2 = 1.0 + 1.0	// = true;	boolean
n) 3 <> 4-1			// = false; boolean
